import { useState, useEffect } from "react";

export default function PassiveIncomeTraining() {
  const [strategies, setStrategies] = useState([]);
  const [search, setSearch] = useState("");
  const [filterRisk, setFilterRisk] = useState("");
  const [filterTag, setFilterTag] = useState("");
  const [filterROI, setFilterROI] = useState("");

  useEffect(() => {
    fetch("/passive_income_strategies.json") // adjust path if needed
      .then(res => res.json())
      .then(setStrategies);
  }, []);

  const filteredStrategies = strategies.filter((s) => {
    const matchesSearch = s.name.toLowerCase().includes(search.toLowerCase());
    const matchesRisk = filterRisk ? s.risk === filterRisk : true;
    const matchesTag = filterTag ? s.tags.includes(filterTag) : true;
    const matchesROI = filterROI ? s.roi_range.includes(filterROI) : true;
    return matchesSearch && matchesRisk && matchesTag && matchesROI;
  });

  const uniqueTags = Array.from(new Set(strategies.flatMap(s => s.tags)));
  const riskLevels = ["Low", "Medium", "High"];
  const roiOptions = ["4", "12", "20", "40", "30", "120"]; // ROI % keywords

  return (
    <div className="p-6 max-w-6xl mx-auto">
      <h1 className="text-4xl font-bold mb-4">ğŸ“š Passive Income Training</h1>

      {/* Search + Filters */}
      <div className="flex flex-wrap gap-4 mb-6">
        <input
          type="text"
          placeholder="ğŸ” Search strategies..."
          className="border p-2 rounded w-full md:w-1/3"
          value={search}
          onChange={(e) => setSearch(e.target.value)}
        />

        <select className="border p-2 rounded" value={filterRisk} onChange={e => setFilterRisk(e.target.value)}>
          <option value="">All Risk Levels</option>
          {riskLevels.map(r => <option key={r} value={r}>{r}</option>)}
        </select>

        <select className="border p-2 rounded" value={filterTag} onChange={e => setFilterTag(e.target.value)}>
          <option value="">All Types</option>
          {uniqueTags.map(t => <option key={t} value={t}>{t}</option>)}
        </select>

        <select className="border p-2 rounded" value={filterROI} onChange={e => setFilterROI(e.target.value)}>
          <option value="">All ROI</option>
          {roiOptions.map(r => <option key={r} value={r}>{`â‰¥ ${r}%`}</option>)}
        </select>
      </div>

      {/* Strategy Cards */}
      {filteredStrategies.map((s) => (
        <div key={s.id} className="border rounded-xl p-4 mb-6 shadow-md bg-white">
          <h2 className="text-2xl font-semibold">{s.name}</h2>
          <p><b>Platform:</b> {s.platform} | <b>Chain:</b> {s.chain} | <b>ROI:</b> {s.roi_range} | <b>Risk:</b> {s.risk}</p>
          <p><b>Tools:</b> {s.tools.join(", ")}</p>

          <details className="mt-3">
            <summary className="cursor-pointer text-blue-600 font-medium">ğŸ“˜ Step-by-Step Guide</summary>
            <ol className="ml-6 mt-2 list-decimal text-sm">{s.steps.map((step, i) => <li key={i}>{step}</li>)}</ol>
          </details>

          <details className="mt-2">
            <summary className="cursor-pointer text-green-600 font-medium">âœ… Exit Strategy</summary>
            <ol className="ml-6 mt-2 list-decimal text-sm">{s.exit_plan.map((step, i) => <li key={i}>{step}</li>)}</ol>
          </details>

            <summary className="cursor-pointer text-red-600 font-medium">âš ï¸ Risk Info</summary>
