Python Snippet: 15-Minute Opening RangeAdd this function to your market_intel.py to handle the mechanical opening strategy.Pythondef get_opening_range(ticker_data):
    # Filter for the 9:30 to 9:45 AM window
    today_open = ticker_data.between_time('09:30', '09:45')
    if not today_open.empty:
        range_high = today_open['High'].max()
        range_low = today_open['Low'].min()
        return range_high, range_low
    return None, None
