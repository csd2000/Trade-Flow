// pages/api/training/[slug].ts
import path from "path";
import fs from "fs";

export default function handler(req, res) {
  try {
    const { slug } = req.query;
    const filePath = path.join(process.cwd(), "data", "training_modules.json");
    const raw = fs.readFileSync(filePath, "utf-8");
    const { modules } = JSON.parse(raw);

    const mod = modules.find((m: any) => m.slug === slug);
    if (!mod) return res.status(404).json({ error: "Module not found" });

    res.status(200).json(mod);
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: "Server error" });
  }
}

// pages/academy/module/[slug].tsx
import { useEffect, useState } from "react";
import { useRouter } from "next/router";
import Link from "next/link";

type ResLink = { label: string; url: string };

type Module = {
  slug: string;
  track: string;
  order: number;
  of: number;
  title: string;
  duration: string;
  overview: string;
  prerequisites: string[];
  steps: string[];
  notes_prompt?: string;
  resources?: {
    video?: ResLink;
    pdf?: ResLink;
    community?: ResLink;
  };
  additional_resources?: ResLink[];
  previous?: { label: string; slug: string } | null;
  next?: { label: string; slug: string } | null;
};

export default function TrainingModule() {
  const router = useRouter();
  const { slug } = router.query;
  const [mod, setMod] = useState<Module | null>(null);
  const [notes, setNotes] = useState("");

  useEffect(() => {
    if (!slug) return;
    fetch(`/api/training/${slug}`)
      .then((r) => r.json())
      .then((data) => {
        setMod(data);
        const saved = localStorage.getItem(`tn-notes-${data.slug}`) || "";
        setNotes(saved);
      })
      .catch(console.error);
  }, [slug]);

  const saveNotes = () => {
    if (!mod) return;
    localStorage.setItem(`tn-notes-${mod.slug}`, notes);
  };

  if (!mod) {
    return (
      <div className="min-h-screen bg-[#1A1F27] text-white p-6">
        <p>Loading module‚Ä¶</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#1A1F27] text-white">
      <div className="max-w-5xl mx-auto p-6">
        <div className="flex items-start justify-between gap-4 flex-wrap">
          <div>
            <h1 className="text-2xl md:text-3xl font-bold">{mod.title}</h1>
            <p className="opacity-80 mt-1">
              {mod.track.replace(/-/g, " ")} ‚Ä¢ {mod.duration} ‚Ä¢ Module {mod.order} of {mod.of}
            </p>
          </div>
          <div className="flex gap-2">
            <button
              onClick={() => window.print()}
              className="rounded-lg px-3 py-2 bg-cyan-600 hover:bg-cyan-500 hover:shadow-lg transition"
            >
              üñ®Ô∏è Print / Save PDF
            </button>
            <Link href="/academy">
              <span className="rounded-lg px-3 py-2 bg-[#1F2630] border border-[#2C3744] hover:bg-[#263140] cursor-pointer transition">
                Back to Academy
              </span>
            </Link>
          </div>
        </div>

        {/* Overview & Prereqs */}
        <div className="grid md:grid-cols-2 gap-4 mt-6">
          <div className="bg-[#232B36] rounded-xl p-4 border border-[#2C3744]">
            <h2 className="text-lg font-semibold mb-2">Overview</h2>
            <p className="opacity-90">{mod.overview}</p>
          </div>
          <div className="bg-[#232B36] rounded-xl p-4 border border-[#2C3744]">
            <h2 className="text-lg font-semibold mb-2">Prerequisites</h2>
            <ul className="list-disc ml-5 space-y-1 text-sm opacity-90">
              {mod.prerequisites?.map((p, i) => <li key={i}>{p}</li>)}
            </ul>
          </div>
        </div>

        {/* Content & Notes */}
        <div className="grid md:grid-cols-2 gap-4 mt-4">
          <div className="bg-[#232B36] rounded-xl p-4 border border-[#2C3744]">
            <h2 className="text-lg font-semibold mb-2">Content ‚Äî Step by Step</h2>
            <ol className="list-decimal ml-5 space-y-1 text-sm opacity-90">
              {mod.steps?.map((s, i) => <li key={i}>{s}</li>)}
            </ol>
          </div>
          <div className="bg-[#232B36] rounded-xl p-4 border border-[#2C3744]">
            <h2 className="text-lg font-semibold mb-2">Notes</h2>
            <p className="text-xs opacity-70 mb-2">{mod.notes_prompt || "Use this space to write your notes."}</p>
            <textarea
              value={notes}
              onChange={(e) => setNotes(e.target.value)}
              className="w-full h-40 rounded-md bg-[#1A2431] border border-[#2C3744] p-2"
              placeholder="Type your notes here‚Ä¶"
            />
            <div className="mt-2 flex gap-2">
              <button onClick={saveNotes} className="rounded-md px-3 py-2 bg-cyan-600 hover:bg-cyan-500">Save Notes</button>
              <button onClick={() => setNotes("")} className="rounded-md px-3 py-2 bg-[#1F2630] border border-[#2C3744] hover:bg-[#263140]">Clear</button>
            </div>
          </div>
        </div>

        {/* Resources */}
        <div className="bg-[#232B36] rounded-xl p-4 border border-[#2C3744] mt-4">
          <h2 className="text-lg font-semibold mb-2">Resources</h2>
          <div className="flex flex-wrap gap-2">
            {mod.resources?.video && (
              <a href={mod.resources.video.url} target="_blank" rel="noreferrer"
                 className="rounded-md px-3 py-2 bg-[#1F2630] border border-[#2C3744] hover:bg-[#263140]">
                üé¨ {mod.resources.video.label}
              </a>
            )}
            {mod.resources?.pdf && (
              <a href={mod.resources.pdf.url} target="_blank" rel="noreferrer"
                 className="rounded-md px-3 py-2 bg-[#1F2630] border border-[#2C3744] hover:bg-[#263140]">
                üìÑ {mod.resources.pdf.label}
              </a>
            )}
            {mod.resources?.community && (
              <a href={mod.resources.community.url} target="_blank" rel="noreferrer"
                 className="rounded-md px-3 py-2 bg-[#1F2630] border border-[#2C3744] hover:bg-[#263140]">
                üí¨ {mod.resources.community.label}
              </a>
            )}
          </div>

          {mod.additional_resources?.length ? (
            <>
              <h3 className="mt-4 font-medium opacity-90">Additional Resources</h3>
              <ul className="list-disc ml-5 space-y-1 mt-1 text-sm">
                {mod.additional_resources.map((r, i) => (
                  <li key={i}><a className="underline" href={r.url} target="_blank" rel="noreferrer">{r.label}</a></li>
                ))}
              </ul>
            </>
          ) : null}
        </div>

        {/* Nav */}
        <div className="flex items-center justify-between mt-6">
          {mod.previous ? (
            <Link href={`/academy/module/${mod.previous.slug}`}><span className="underline cursor-pointer">‚Üê {mod.previous.label}</span></Link>
          ) : <span className="opacity-40">Start</span>}
          {mod.next ? (
            <Link href={`/academy/module/${mod.next.slug}`}><span className="underline cursor-pointer">{mod.next.label} ‚Üí</span></Link>
          ) : <span className="opacity-40">End</span>}
        </div>
      </div>

      <style jsx global>{`
        @media print {
          button, .cursor-pointer, a { display: none; }
          body { background: #fff !important; color: #111 !important; }
        }
      `}</style>
    </div>
  );
}
