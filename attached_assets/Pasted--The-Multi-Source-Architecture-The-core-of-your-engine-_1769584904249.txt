class CentralScanner:
    def __init__(self, symbol):
        self.symbol = symbol
        self.gates = {
            "TV_Liquidity_Detected": False,  # Source: TradingView
            "MT5_Volume_Spike": False,      # Source: MetaTrader 5
            "AI_Fakeout_Prob": 0.0          # Source: Replit AI Logic
        }

    def process_sweep(self, price_data):
        # GATE 1: PRIMARY TRIGGER (Liquidity Sweep)
        # Check if current low < previous swing low AND close > previous swing low
        if price_data['low'] < price_data['swing_low'] and price_data['close'] > price_data['swing_low']:
            
            # GATE 2: SOURCE VALIDATION
            if self.gates["TV_Liquidity_Detected"] and self.gates["MT5_Volume_Spike"]:
                
                # GATE 3: AI CONFIDENCE
                if self.gates["AI_Fakeout_Prob"] > 0.75:
                    return "EXECUTE_LONG_SNIPER"
        
        return "WAITING_FOR_SWEEP"

The Trap: Price spikes below a known Swing Low (Retail stops triggered).

The Trigger: This is the only time the scanner "wakes up" for a trade.
