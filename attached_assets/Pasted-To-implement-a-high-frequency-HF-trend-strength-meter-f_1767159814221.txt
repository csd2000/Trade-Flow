To implement a high-frequency (HF) trend strength meter for all assets, your backend needs to compute Relative Momentum and Trend Energy in a loop, while your frontend displays this through a dynamic bar meter.

1. The Multi-Asset Trend Engine (Python)
This script calculates the Trend Strength Score (0-100) for your entire watchlist. It uses the ADX to measure trend intensity and the EMA Slope to determine direction and velocity.

Python

import pandas as pd
import pandas_ta as ta
import yfinance as yf

# Expand this list to include all assets in your app
watchlist = ["XAUUSD=X", "XAGUSD=X", "EURUSD=X", "BTC-USD", "ETH-USD"]

def get_trend_meter_data(assets):
    payload = []
    for ticker in assets:
        # Fetch 1-minute data for scalping precision
        df = yf.download(ticker, interval="1m", period="1d", progress=False)
        
        # 1. Trend Energy (ADX)
        adx_df = ta.adx(df['High'], df['Low'], df['Close'], length=14)
        current_adx = adx_df['ADX_14'].iloc[-1]
        
        # 2. Trend Velocity (EMA Slope)
        df['EMA9'] = ta.ema(df['Close'], length=9)
        # Calculate angle of the 9 EMA over last 3 bars
        slope = (df['EMA9'].iloc[-1] - df['EMA9'].iloc[-3]) / 3
        
        # 3. Strength Scoring (0-100)
        # ADX > 25 is trending; we normalize 0-50 ADX to a 0-100 scale
        strength_score = min(int(current_adx * 2), 100)
        
        # 4. Color Logic
        is_bullish = slope > 0
        
        payload.append({
            "ticker": ticker,
            "score": strength_score,
            "is_bullish": is_bullish,
            "slope": round(slope, 6)
        })
    return payload

# Output example: [{'ticker': 'XAUUSD=X', 'score': 64, 'is_bullish': True, ...}]
