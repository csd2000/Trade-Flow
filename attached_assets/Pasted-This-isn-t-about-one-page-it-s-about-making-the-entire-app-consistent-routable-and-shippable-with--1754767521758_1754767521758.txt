7) Sanity checks (catch issues before build)
Add a tiny script to guarantee unique ids/slugs and route health.

client/scripts/checkStrategies.ts

ts
Copy
Edit
import { strategies } from "../src/data/strategies";

const ids = new Set<number>();
const slugs = new Set<string>();
let ok = true;

for (const s of strategies) {
  if (ids.has(s.id)) { console.error("Duplicate id:", s.id); ok = false; }
  ids.add(s.id);
  if (slugs.has(s.slug)) { console.error("Duplicate slug:", s.slug); ok = false; }
  slugs.add(s.slug);
}

if (!ok) process.exit(1);
console.log("Strategies OK:", strategies.length);
